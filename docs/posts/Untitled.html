import numpy as np
class Base:
  def __init__(self, end_condition, name):
    self.end_condition = end_condition
    self.name = name
    self.top = 0
    self.n=0
    self.finish = False
  def __repr__(self):
    text=(
        f'이름: {self.name}\n'
        f'목표높이: {self.end_condition}\n'
        f'현재높이: {self.top}\n'
        f'진행횟수: {self.n}\n'
        f'완성상태: {self.finish}'
        )
    return text
  def 탑쌓기(self):
    self.n += 1
    if self.top < self.end_condition:
      self.top += 1
      if self.top == self.end_condition:
        self.finish = True
  def __gt__(self, other):
    return self.top > other.top
  def __lt__(self, other):
    return self.top < other.top
  def __rshift__(self, other):
    if self.finish==True :
      return True
    else:
      return False
  def __lshift__(self, other):
    if other.finish==True:
      return True
    else:
      return False

    self.n += 1
    other.n +=1

      egyo=np.random.choice(['suc','fail'], p=[0.2, 0.8])
      if egyo=='suc':
        suc_or_fail=np.random.choice([2,0], p=[self.prob, 1-self.prob])
        if suc_or_fail:
          self.top += suc_or_fail
          return('애교성공->탑쌓기성공')
        else:
          return('애교성공->탑쌓기실패')
      if egyo== 'fail':
        suc_or_fail=np.random.choice([2,0], p=[other.prob, 1-other.prob])
        if suc_or_fail:
          other.top += suc_or_fail
          return('애교실패->탑쌓기성공')
        else:
          other.top=0
          return('애교실패->탑쌓기실패')
    if  self.top >= self.end_condition :
      self.finish=True
    if  other.top >= other.end_condition :
      other.finish=True
